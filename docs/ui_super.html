getProto (method)

contr/view/model/collection.getProto()

Returns the first object up above the prototype chain

params

None

returns

Object

example usage

    (function (Zonzabone, options) {
        "use strict";

        return function (options) {

            var that = Object.create(Zonzabone.controller(options));

            that.init = function (options) {
                this.getProto().init.bind(this)(options);

                return this;
            };

            return that.init(options);
        };
    }(Zonzabone, options));
